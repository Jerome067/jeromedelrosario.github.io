⚙️ Calculator - C++

A simple calculator implementation in C++

About this Implementation

This is a command-line calculator written in C++ that demonstrates fundamental programming concepts including functions, loops, and conditional statements. It includes proper error handling and supports all basic arithmetic operations.

Source Code

// Simple Calculator in C++
// Author: Jerome M. Del Rosario

#include <iostream>
using namespace std;

double add(double a, double b) {
    return a + b;
}

double subtract(double a, double b) {
    return a - b;
}

double multiply(double a, double b) {
    return a * b;
}

double divide(double a, double b) {
    if (b == 0) {
        cout << "Error: Division by zero!" << endl;
        return 0;
    }
    return a / b;
}

double modulo(int a, int b) {
    if (b == 0) {
        cout << "Error: Division by zero!" << endl;
        return 0;
    }
    return a % b;
}

int main() {
    double num1, num2;
    char operation;
    char continueCalc;
    
    cout << "===== Simple Calculator =====" << endl;
    cout << "Operations: +, -, *, /, %" << endl;
    cout << "==============================" << endl;
    
    do {
        cout << "\nEnter first number: ";
        cin >> num1;
        
        cout << "Enter operation (+, -, *, /, %): ";
        cin >> operation;
        
        cout << "Enter second number: ";
        cin >> num2;
        
        switch(operation) {
            case '+':
                cout << "Result: " << add(num1, num2) << endl;
                break;
            case '-':
                cout << "Result: " << subtract(num1, num2) << endl;
                break;
            case '*':
                cout << "Result: " << multiply(num1, num2) << endl;
                break;
            case '/':
                cout << "Result: " << divide(num1, num2) << endl;
                break;
            case '%':
                cout << "Result: " << modulo((int)num1, (int)num2) << endl;
                break;
            default:
                cout << "Invalid operation!" << endl;
        }
        
        cout << "Calculate again? (y/n): ";
        cin >> continueCalc;
        
    } while(continueCalc == 'y' || continueCalc == 'Y');
    
    cout << "Thank you for using Calculator!" << endl;
    
    return 0;
}
